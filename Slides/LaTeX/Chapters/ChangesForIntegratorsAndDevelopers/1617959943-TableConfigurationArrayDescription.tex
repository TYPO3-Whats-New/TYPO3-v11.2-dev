% Feature | 93794 | Override TCA description with TSconfig

\begin{frame}[fragile]
	\frametitle{Changes for Integrators and Developers}
	\framesubtitle{TCA Description (1)}

	\begin{itemize}
		\item Since TYPO3 v9, every TCA field can have a description next to
			its label. For example:
	\end{itemize}
	\begin{figure}
		\includegraphics[width=0.70\linewidth]{ChangesForIntegratorsAndDevelopers/1617959943-TableConfigurationArrayDescription}
	\end{figure}
	\begin{itemize}
		\item Integrators can now overwrite the description by using Page TSconfig
	\end{itemize}
\end{frame}

% ------------------------------------------------------------------------------

\begin{frame}[fragile]
	\frametitle{Changes for Integrators and Developers}
	\framesubtitle{TCA Description (2)}

	\lstset{basicstyle=\fontsize{8}{10}\ttfamily}

	\begin{itemize}
		\item This option can also be used to set a description for a field
			that does not feature a TCA description yet
	\end{itemize}

	\begin{itemize}
		\item Examples:
		\begin{itemize}
			\item Example 1 - set/overwrite a description:
\begin{lstlisting}
TCEFORM.<table>.<field>.description = The headline is the text...
\end{lstlisting}
		\end{itemize}

		\begin{itemize}
			\item Example 2 - on a per record type basis:
\begin{lstlisting}
TCEFORM.<table>.<field>.types.<type>.description = Description for <type>
\end{lstlisting}
		\end{itemize}
	\end{itemize}

\end{frame}
