% Feature | 93209 | FAL: Add getFile() to TYPO3/CMS/Core/Resource/Folder

\begin{frame}[fragile]
	\frametitle{Changes for Integrators and Developers}
	\framesubtitle{FAL Folder Object}

	\lstset{basicstyle=\fontsize{8}{10}\ttfamily}

	\begin{itemize}
		\item The FAL \texttt{Folder} object now contains a convenience method
			to retrieve a file\newline
			\smaller\textbf{Old approach:}\normalsize
\begin{lstlisting}
$folder->getStorage()->getFileInFolder("filename.ext", $folder)
\end{lstlisting}

		\item \smaller\textbf{New approach available since TYPO3 v11.2}:\normalsize
\begin{lstlisting}
$folder->getFile("filename.ext")
\end{lstlisting}

	\end{itemize}
\end{frame}
